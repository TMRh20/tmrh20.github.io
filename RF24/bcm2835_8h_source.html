643 // Defines for ST
644 // GPIO register offsets from BCM2835_ST_BASE.
645 // Offsets into the ST Peripheral block in bytes per 12.1 System Timer Registers
646 // The System Timer peripheral provides four 32-bit timer channels and a single 64-bit free running counter.
647 // BCM2835_ST_CLO is the System Timer Counter Lower bits register.
648 // The system timer free-running counter lower register is a read-only register that returns the current value
649 // of the lower 32-bits of the free running counter.
650 // BCM2835_ST_CHI is the System Timer Counter Upper bits register.
651 // The system timer free-running counter upper register is a read-only register that returns the current value
652 // of the upper 32-bits of the free running counter.
653 #define BCM2835_ST_CS 0x0000
654 #define BCM2835_ST_CLO 0x0004
655 #define BCM2835_ST_CHI 0x0008
